SELECT Orders.orderid, customers.customername, orders.orderdate, products.productname, categories.categoryname FROM orders JOIN customers ON orders.customerid = customers.customerid JOIN orderdetails ON orders.orderid = orderdetails.orderid JOIN products ON orderdetails.productid = products.productid JOIN categories ON products.categoryid = categories.categoryid WHERE orders.shipdate BETWEEN (CURRENT_DATE - interval '1 month') AND CURRENT_DATE ORDER BY orders.orderid NULLS LAST;
