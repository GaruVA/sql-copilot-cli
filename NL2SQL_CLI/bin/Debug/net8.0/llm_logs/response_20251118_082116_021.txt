SELECT p.productid, c.categoryname, SUM(od.discount * od.unitprice) AS revenue FROM products p JOIN orderdetails od ON p.productid = od.productid JOIN categories c ON p.categoryid = c.categoryid WHERE od.discount > 0.5 GROUP BY p.productid, c.categoryname ORDER BY revenue DESC NULLS LAST LIMIT 5;
